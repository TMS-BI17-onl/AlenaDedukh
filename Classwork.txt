SELECT Person.Person.FirstName,  Person.Person.LastName, Person.PersonPhone.PhoneNumber
FROM Person.Person
JOIN Person.PersonPhone
ON Person.Person.BusinessEntityID=Person.PersonPhone.BusinessEntityID
WHERE Person.PersonPhone.PhoneNumber LIKE '4%5'

CASE WHEN HumanResoueses.Employee.Age BETWEEN 17 AND 20 THEN 'Adolescence'
	WHEN HumanResoueses.Employee. BETWEEN 21 AND 59 THEN 'Adults'
	WHEN HumanResoueses.Employee. BETWEEN 60 AND 75 THEN 'Elderly'
	WHEN HumanResoueses.Employee. BETWEEN 76 AND 90 THEN 'Senile'
	ELSE 'No Category'
